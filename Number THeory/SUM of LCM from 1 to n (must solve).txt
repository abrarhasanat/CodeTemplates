 /// idea :- https://forthright48.com/spoj-lcmsum-lcm-sum ///
ll ans[mx];
ll phi[mx];
void precal(){
    For(i,mx) phi[i] =i;
    for(int i = 2; i <= mx; ++i){
        if(phi[i] == i){
            phi[i]  = i-1;
            for(int j = 2*i ; j <= mx ;j +=i){
                phi[j] -= (phi[j] / i);
            }
        }
    }
    ans[1] = 1;
    for(int i = 1; i <= mx ; ++i){
        for(int j = i ; j <= mx; j += i){
            ans[j]+= (phi[i] * i);
        }
    }
}
int main(){
    int t;
    scanf("%d", &t);
    precal();

        scanf("%d", &n);
        ll res = ans[n];
        res = n *  (1 + res);

    }
}
